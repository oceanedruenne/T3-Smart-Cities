91 tilesd = new Dictionary<Vector3Int, GameObject>();
92
93 int posX = 0;
94 int posY = 0;
95 int posZ = 0;
96
97 for (posX = 0; posX < size; posX++)
98 {
99 for(posY = 0; posY < size; posY++)
100 {
101 GameObject tile = Instantiate(tilePrefab, grid.transform);
102
103 tile.transform.parent = gridObject.transform;
104 tile.transform.position = grid.CellToWorld(new Vector3Int(posX, posY, posZ));
105 tile.name = posX + " , " + posY;
106
107 tilesd[new Vector3Int(posX, posY, posZ)] = tile;
108 }
109 }
